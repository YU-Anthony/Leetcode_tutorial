class Solution:
    """
    dp[i] = the number of ways to parse string s[1 : i + 1]
    """
    def numDecodings(self, s: str) -> int:
        dp = [0]*(len(s) + 1)
        dp[0] = 1
        dp[1] = 0 if s[0] == '0' else 1
        
        for i in range(2, len(s) + 1):
            if 0 < int(s[i - 1: i]) <= 9:
                dp[i] += dp[i - 1]
            if 10 <= int(s[i -2:i]) <=26:
                dp[i] += dp[i - 2]
        return dp[-1]
